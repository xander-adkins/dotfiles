* Emacs Configuration
This is my custom emacs configuration whitten as an Org Mode doc. This file generates [[file:init.el][init.el]] which is loaded by emacs at startup.

#+BEGIN_SRC emacs-lisp

  ;; NOTE: init.el is now generated from Emacs.org.  Please edit that file
  ;;       in Emacs and init.el will be generated automatically!

#+END_SRC

* Keep init.el Tidy
Don't let custom-set-variables dirty the init.el file.

#+BEGIN_SRC emacs-lisp

  (setq custom-file (concat user-emacs-directory "/custom.el"))

#+END_SRC

* System Settings
#+BEGIN_SRC emacs-lisp

  (prefer-coding-system 'utf-8)                                                 ; Use utf-8 everywhere
  (set-default-coding-systems 'utf-8)
  (set-terminal-coding-system 'utf-8)
  (set-keyboard-coding-system 'utf-8)
  (setq default-buffer-file-coding-system 'utf-8)

  (setq select-enable-clipboard t)                                              ; Cut & paste with the system clipboard
  (setq-default indent-tabs-mode nil)                                           ; Keep spaces and don't default to tab insertion

  (setq undo-limit 6710886400)                                                  ; Increase undo limit to 64MB
  (defalias 'yes-or-no-p 'y-or-n-p)                                             ; Answer 'y' or 'n' for text-mode prompts


#+END_SRC

* Reset UI Configuration
Configure basic UI settings to remove unwanted emacs features

#+BEGIN_SRC emacs-lisp

  (setq inhibit-startup-message t) 						                        ; Disable startup screen
  (tool-bar-mode -1)                                                            ; Disable the toolbar
  (tooltip-mode -1)                                                             ; Disable tooltips
  (menu-bar-mode -1)                                                            ; Disable the menu bar
  (setq visible-bell 1)                                                         ; Disable visual bell sound
  (setq bidi-display-reordering nil)                                            ; Disable bidirectional text for performance
  (setq make-pointer-invisible t)                                               ; Hide mouse cursor while typing
  (setq cursor-in-non-selected-windows 'hollow)                                 ; Only show cursor in selected window

#+END_SRC

* Package System 
Make use of [[https://github.com/jwiegley/use-package][use-package]] to automate a lot of the package setup required in Emacs default package manager.

#+BEGIN_SRC emacs-lisp

  (require 'package)                                                            ; Initialize package sources 
  (setq package-archives '(("melpa" . "https://melpa.org/packages/")
                         ("elpa". "https://elpa.gnu.org/packages/")
                         ("org" . "https://orgmode.org/elpa/")))                ; For org mode packages

  (package-initialize)                                                          ; Initializes the package system
  (unless package-archive-contents
    (package-refresh-contents))                                                 ; Update the package list

  (unless (package-installed-p 'use-package)                                    ; Auto-install use-package 
    (package-install 'use-package))                                             ; Use default package-install to install use-package

  (require 'use-package)                                                        ; Auto-install use-package
  (setq use-package-always-ensure t)                                            ; Download and install packages automatically
  (setq use-package-always-defer t)                                             ; Defer loading mode or key-binding specific packages

#+END_SRC

* Formatting & Syntax Highlighting Configurations 

#+BEGIN_SRC emacs-lisp

  (use-package glsl-mode)                                                       ; Glsl language support

  (use-package lsp-mode                                                         ; Language Server Protocol Support for Emacs 
    :init
      (setq lsp-keymap-prefix "C-c l")                                          ; Set prefix for lsp-command-keymap
    :hook (typescript-mode . lsp)
    :commands lsp)

  (use-package markdown-mode                                                    ; Markdown language support
    :hook (markdown-mode . auto-fill-mode))                                     ; Automatically break lines of buffer content

  (use-package tree-sitter                                                      ; Tree-sitter syntax highlighting 
  :config
    (require 'tree-sitter-langs)
    (global-tree-sitter-mode)
    (add-hook 'tree-sitter-after-on-hook #'tree-sitter-hl-mode))

  (use-package typescript-mode                                                  ; Typescript language support
    :mode ("\\.tsx?\\'" . typescript-mode)
    :config
    (setq typescript-indent-level 2))

#+END_SRC

* Keybinding Configuration
This configuration is heavily dependent on [[https://evil.readthedocs.io/en/latest/index.html][evil-mode]]. 

#+BEGIN_SRC emacs-lisp

  (global-set-key (kbd "<escape>") 'keyboard-escape-quit)                       ; Make ESC quit prompts

  (use-package undo-fu)                                                         ; Package for sensible redo support

  (use-package evil                                                             ; Evil VIM key bindings
    :demand t
    :init
    (setq evil-undo-system 'undo-fu)                                            ; Set evil undo to undo-fu package
    :config                                                                     ; Initialize evil-mode
    (evil-mode 1)                                                               ; Initiate Evil Mode on startup
    (define-key evil-insert-state-map (kbd "C-g") 'evil-normal-state)           ; Bind Emacs "quit" key to evil normal mode
    (evil-global-set-key 'motion "j" 'evil-next-visual-line)                    ; Next line includes wrapped lines
    (evil-global-set-key 'motion "k" 'evil-previous-visual-line)                ; Previous line includes wrapped lines
    (evil-global-set-key 'motion "C-l" 'evil-window-right)                      ; Move to right window 
    (evil-global-set-key 'motion "C-h" 'evil-window-left))                      ; Move to left window 

  (use-package general                                                          ; More convenient key definitions
    :config
    (general-evil-setup t)
    (general-create-definer aa/leader-keys
      :keymaps '(normal visual insert emacs)
      :prefix "SPC"
      :global-prefix "C-SPC"))


#+END_SRC
